# 1. Crear la Service Account
gcloud iam service-accounts create terraform-admin \
    --display-name="Terraform Admin Account"

# 2. Darle el rol de Editor (o Owner) en el proyecto
gcloud projects add-iam-policy-binding YOUR_PROJECT_ID \
    --member="serviceAccount:terraform-admin@YOUR_PROJECT_ID.iam.gserviceaccount.com" \
    --role="roles/editor"

# 3. PERMISO CLAVE: Permitir que TU usuario personal pueda suplantar a esta cuenta
# Sustituye YOUR_EMAIL@gmail.com por tu cuenta de Google
gcloud iam service-accounts add-iam-policy-binding \
    terraform-admin@YOUR_PROJECT_ID.iam.gserviceaccount.com \
    --member="user:YOUR_EMAIL@gmail.com" \
    --role="roles/iam.serviceAccountTokenCreator"
